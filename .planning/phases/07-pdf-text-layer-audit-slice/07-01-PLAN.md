# src/inference/pdf_loader.py (to be created in Wave 1, asserted in Wave 0)

```python
class PdfNoTextLayer(Exception):
    """Raised when a PDF has fewer than 50% pages with extractable text."""

def check_text_layer(path: str | Path) -> float:
    """Return ratio of pages with non-empty text in [0.0, 1.0]; 0.0 for 0-page PDF."""

def extract_pdf_blocks(path: str | Path) -> list[dict]:
    """Return report-row dicts; one per text block + one sentinel per image-only page.
    Each row dict has keys: block_id (str 'p{page_no}b{block_no}', page_no 1-based,
    block_no 0-based from fitz), kind ('paragraph'), text (str ≤500), status ('review'),
    applied_fixes (''), explanation (str)."""
```
