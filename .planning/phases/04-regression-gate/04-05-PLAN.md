on:
  pull_request: branches [main, master]
  push: branches [main, master]

env:
  CI: "true"     # explicit (GHA also auto-sets it; this makes Pitfall 5 fire deterministically)

    - run: pytest -q \
        tests/test_negative_corpus_diff_rate.py \
        tests/test_positive_docx_regression.py \

        tests/test_format_regression_audit.py    # NOT `make` — ubuntu-latest may not have GNU Make

    ```yaml
    name: regression-gate

    on:
      pull_request:
        branches: [main, master]
      push:
        branches: [main, master]

    env:
      CI: "true"

    jobs:
      gate:
        runs-on: ubuntu-latest

        steps:

          - uses: actions/setup-python@v5
            with:

              cache: 'pip'
          - name: Install dependencies
            run: pip install -r requirements.txt
          - name: Run regression gate
            run: |
              pytest -q \
                tests/test_negative_corpus_diff_rate.py \
                tests/test_positive_docx_regression.py \
                tests/test_rules_quality_acceptance.py \
                tests/test_format_regression_audit.py
    ```
